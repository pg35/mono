interface Config {
  enable: boolean;
  type: "discount" | "surcharge";
  valueType: "fixed" | "percent";
  value: number;
}

type State = [
  {
    "*": Config;
    guest: Config;
  },
  {
    [role: string]: Config;
  },
];
const defaultConfig: Config = {
  enable: false,
  type: "discount",
  valueType: "fixed",
  value: 0.0,
};
const initialState: State = [
  {
    "*": defaultConfig,
    guest: defaultConfig,
  },
  {},
];

export function RadioGroupDemo() {

}

function App() {
  const [state, setState] = useState<State>(initialState);
